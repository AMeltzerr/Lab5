AdapterPattern.java 
// Interfață existentă
interface StandardCharger {
    void charge();
}

// Clasă incompatibilă
class EuropeanPlug {
    void connect() {
        System.out.println("Conectat la priză europeană.");
    }
}

// Adapter care face EuropeanPlug compatibil cu StandardCharger
class PlugAdapter implements StandardCharger {
    private EuropeanPlug plug;

    public PlugAdapter(EuropeanPlug plug) {
        this.plug = plug;
    }

    public void charge() {
        plug.connect();
    }
}
